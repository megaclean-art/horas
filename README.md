function toFS(data) {
  const fields = {};
  for (const [k, v] of Object.entries(data)) {
    if (v === null || v === undefined) fields[k] = { nullValue: null };
    else if (typeof v === 'boolean') fields[k] = { booleanValue: v };
    else if (typeof v === 'number') fields[k] = { doubleValue: v };
    else if (typeof v === 'string') fields[k] = { stringValue: v };
    else if (Array.isArray(v)) fields[k] = { arrayValue: { values: v.map(i => ({ stringValue: String(i) })) } };
    else fields[k] = { stringValue: String(v) };
  }
  return { fields };
}

async function fsSet(col, id, data) {
  const mask = Object.keys(data).map(k => 'updateMask.fieldPaths=' + k).join('&');
  const r = await fetch(API + '/' + col + '/' + id + '?' + mask, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(toFS(data))
  });
  return r.ok;
}

async function fsAdd(col, data) {
  const r = await fetch(API + '/' + col, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(toFS(data))
  });
  const d = await r.json();
  return d.name?.split('/').pop();
}

      setProgress((i / DATA.operarios.length) * 40);
      await sleep(100);
    }
  }
  log('✓ ' + DATA.operarios.length + ' operarios creados', 'ok');

  // PASO 2: Importar servicios
  log('--- PASO 2: Creando servicios ---', 'info');
  let j = 0;
  for (const s of DATA.servicios) {
    const opId = operarioIds[s.operarioNombre] || '';
    const id = 'srv_' + s.codigo + '_' + (s.operarioNombre.replace(/[^a-zA-Z0-9]/g, '_').toLowerCase().slice(0,20));
    const ok = await fsSet('horas_servicios', id, {
      codigo: s.codigo,
      cliente: s.cliente,
      supervisor: s.supervisor,
      direccion: s.direccion,
      operarios: [opId],
      operarioNombre: s.operarioNombre,
      dias: s.dias,
      horaInicio: s.horaInicio,
      horaFin: s.horaFin,
      horasContratadas: s.horasContratadas,
      tipo: s.tipo,
      radioGps: s.radioGps,
      lat: s.lat,
      lon: s.lon
    });
    j++;
    if (j % 20 === 0 || j === DATA.servicios.length) {
      log('Servicios: ' + j + '/' + DATA.servicios.length, ok ? 'ok' : 'err');
      setProgress(40 + (j / DATA.servicios.length) * 60);
      await sleep(150);
    }
  }

  log('✓ ' + DATA.servicios.length + ' servicios creados', 'ok');
  log('', 'info');
  log('✅ IMPORTACIÓN COMPLETA', 'ok');
  log('Ahora podés cerrar esta página y abrir la app principal.', 'info');
  setProgress(100);
  document.getElementById('btn-import').textContent = '✓ IMPORTACIÓN COMPLETADA';
  document.getElementById('btn-clear').style.display = 'block';
  document.querySelector('.card').classList.add('done');
}

async function clearAll() {
  if (!confirm('¿Borrar TODOS los servicios y operarios importados de Firebase? Esto no se puede deshacer.')) return;
  log('Borrando datos...', 'warn');
  // List and delete
  const cols = ['horas_servicios', 'horas_usuarios'];
  for (const col of cols) {
    const r = await fetch(API + '/' + col + '?pageSize=300');
    const d = await r.json();
    if (!d.documents) { log('Sin documentos en ' + col, 'warn'); continue; }
    for (const doc of d.documents) {
      await fetch(doc.name.replace('https://firestore.googleapis.com/v1/', 'https://firestore.googleapis.com/v1/'), { method: 'DELETE' });
    }
    log('Borrados: ' + d.documents.length + ' en ' + col, 'ok');
  }
  log('Listo. Podés reimportar.', 'info');
}
